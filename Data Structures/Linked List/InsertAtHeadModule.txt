  class Node {

    // This is a "method-only" submission.

Node InsertNth(Node head, int data, int position) {

    Node p = head;
    Node q = new Node();
    q.data = data;
    if(position == 0){   
        q.next = head;
        return q;
    }
    else{
        for( int i = 0; i < position-1; i++){
            p = p.next;
        }
        q.next = p.next;
        p.next = q;
        return head;
    }
}
